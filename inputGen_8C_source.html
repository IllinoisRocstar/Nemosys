104 {
105  bool onlyKey = false;
106  if (!key.empty())

108  for (const auto &it : _inp)
109  {

111  continue;
112

114  std::ofstream ofile;
115  ofile.open(fname);
116  if (!ofile.good())
117  {
118  std::cerr << "Error opening file " << fname << std::endl;
119  throw;
120  }
121  ofile << (it.second)->str();
122  ofile.close();
123  }
