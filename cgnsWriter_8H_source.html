67  if (mode == 0) {

69  &indexFile))
70  cg_error_exit();
71  } else if (mode == 1) {
72  if (cg_open(myCgFileName.c_str(), CG_MODE_MODIFY,
73  &indexFile))
74  cg_error_exit();
75  }
