50  if (this->files_.inputMeshFile.find(".vt") != std::string::npos) {
51  std::cout << "Detected file in VTK format" << std::endl;

53  } else {
54  std::cerr << "Source mesh file is not in VTK format" << std::endl;
55  }
56 
57  std::ifstream meshStream(this->files_.inputMeshFile);
58  if (!meshStream.good()) {

61  exit(1);
62  }
63  // create meshBase object

66
